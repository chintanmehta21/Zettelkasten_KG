# RAG E2E Evaluation & Module Refactor Loop — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Wire the RAG chunk ingest into the live website capture path, build a 120-question evaluation harness that scores answers on a 4-axis rubric, and run a module-by-module refactor loop that stops at mean ≥17/20 or 20 total iterations.

**Architecture:**
Phase A finishes the refactor started in the previous session — move `_maybe_ingest_rag_chunks` out of `nexus/service/persist.py` into the canonical `website.features.rag_pipeline.ingest.hook` module, migrate its tests, flip `rag_chunks_enabled=true`, and verify one real Naruto capture produces `kg_node_chunks` rows on prod Supabase.
Phase B builds an in-tree evaluation package at `website/features/rag_pipeline/evaluation/` with a rubric B judge (4-axis, 0–5) running through Claude Sonnet via Anthropic API directly (separate from the Gemini key pool), a rubric C RAGAS scaffold, a deterministic corpus clusterer for Naruto's 34 zettels, a 120-Q dataset stored in YAML, and a scorecard generator that emits per-iteration markdown reports.
Phase C dispatches to `superpowers:subagent-driven-development` for the module refactor loop: one module per batch of 3–5 iterations, priority order ingest → retrieval+query → rerank → context → generation+critic, each iteration refactor→deploy→eval→score with a hard global cap of 20 iterations.

**Tech Stack:** Python 3.12, FastAPI, Pydantic, Supabase (Postgres + pgvector), Gemini embeddings via `GeminiKeyPool`, Anthropic Claude Sonnet for judging (separate API key), pytest + pytest-asyncio, PyYAML, httpx, DigitalOcean droplet blue/green deploy.

"""Shared pydantic models for the evaluation harness."""

from __future__ import annotations

from datetime import datetime
from typing import Literal

from pydantic import BaseModel, Field

Category = Literal[
    "single_zettel_factual",
    "multi_zettel_synthesis",
    "cross_source",
    "temporal",
    "out_of_corpus",
    "adversarial",
    "numeric",
    "negation",
]


class Kasten(BaseModel):
    """A topic cluster of 5-7 zettels."""

    id: str
    title: str
    zettel_ids: list[str]
    dominant_tags: list[str]


class Question(BaseModel):
    id: str
    category: Category
    kasten_id: str
    question: str
    expected_zettel_ids: list[str] = Field(default_factory=list)
    expected_behavior: Literal["answer", "refuse", "partial"] = "answer"


class AxisScore(BaseModel):
    faithfulness: int = Field(..., ge=0, le=5)
    relevance: int = Field(..., ge=0, le=5)
    completeness: int = Field(..., ge=0, le=5)
    citation_accuracy: int = Field(..., ge=0, le=5)

    @property
    def total(self) -> int:
        return (
            self.faithfulness
            + self.relevance
            + self.completeness
            + self.citation_accuracy
        )


class JudgeResult(BaseModel):
    scores: AxisScore
    rationale: dict[str, str]


class QuestionResult(BaseModel):
    question_id: str
    category: Category
    kasten_id: str
    question: str
    answer: str
    retrieved_chunk_ids: list[str]
    latency_ms: int
    judge: JudgeResult | None = None
    error: str | None = None


class RunReport(BaseModel):
    iteration: int
    module_under_test: str
    started_at: datetime
    finished_at: datetime
    git_sha: str
    rubric: Literal["b", "c"]
    results: list[QuestionResult]

    def mean_total(self) -> float:
        scored = [r for r in self.results if r.judge is not None]
        if not scored:
            return 0.0
        return sum(r.judge.scores.total for r in scored) / len(scored)
```

from .types import Kasten

_TAG_NORMALIZE_RE = re.compile(r"[^a-z0-9/]+")


def _normalize_tag(tag: str) -> str:
    return _TAG_NORMALIZE_RE.sub("-", tag.lower()).strip("-")


def jaccard_similarity(a: set[str], b: set[str]) -> float:
    if not a and not b:
        return 0.0
    intersection = len(a & b)
    union = len(a | b)
    if union == 0:
        return 0.0
    return intersection / union


def _title_bigrams(title: str) -> set[str]:
    tokens = re.findall(r"[a-z0-9]+", title.lower())
    return {f"{tokens[i]} {tokens[i+1]}" for i in range(len(tokens) - 1)}


def _node_signature(node: dict) -> set[str]:
    tags = {_normalize_tag(t) for t in (node.get("tags") or []) if t}
    tags.discard("")
    if tags:
        return tags
    return _title_bigrams(str(node.get("name") or ""))


def cluster_into_kastens(
    nodes: Iterable[dict],
    *,
    target_kastens: int = 6,
    min_size: int = 5,
    max_size: int = 7,
    similarity_threshold: float = 0.15,
) -> list[Kasten]:
    """Group nodes into topic-coherent Kastens.

    Deterministic: output is purely a function of the input node list (sorted by id).
    Greedy algorithm — picks the node with the most tag overlap with the current
    seed, then iterates. Not optimal but stable across runs, which is what the
    eval loop requires.
    """

    sorted_nodes = sorted(list(nodes), key=lambda n: str(n.get("id") or ""))
    remaining = {str(n["id"]): n for n in sorted_nodes}
    kastens: list[Kasten] = []
    kasten_index = 0

    while remaining and len(kastens) < target_kastens:
        seed_id = next(iter(remaining))
        seed_node = remaining.pop(seed_id)
        seed_sig = _node_signature(seed_node)

        members = [seed_node]
        member_sig = set(seed_sig)

        while len(members) < max_size:
            best_id: str | None = None
            best_score = similarity_threshold
            for cand_id, cand_node in remaining.items():
                cand_sig = _node_signature(cand_node)
                score = jaccard_similarity(member_sig, cand_sig)
                if score > best_score or (score == best_score and best_id is None):
                    best_id = cand_id
                    best_score = score
            if best_id is None:
                break
            picked = remaining.pop(best_id)
            members.append(picked)
            member_sig |= _node_signature(picked)

        if len(members) < min_size and remaining:
            for member in members:
                remaining[str(member["id"])] = member
            break

        dominant = sorted(member_sig)[:5]
        kasten_index += 1
        kastens.append(
            Kasten(
                id=f"kasten-{kasten_index:02d}",
                title=_kasten_title(members, dominant),
                zettel_ids=[str(m["id"]) for m in members],
                dominant_tags=dominant,
            )
        )

    return kastens


def _kasten_title(members: list[dict], dominant: list[str]) -> str:
    if dominant:
        return " / ".join(dominant[:3])
    names = [str(m.get("name") or "") for m in members[:2]]
    return " / ".join(n for n in names if n) or "untitled-kasten"
```

from __future__ import annotations

from pathlib import Path

import yaml

from .types import Category, Question

_EXPECTED_COUNT = 120
_EXPECTED_CATEGORIES: tuple[Category, ...] = (
    "single_zettel_factual",
    "multi_zettel_synthesis",
    "cross_source",
    "temporal",
    "out_of_corpus",

    "numeric",
    "negation",
)

def load_questions(path: Path) -> list[Question]:
    """Load and strictly validate the questions.yaml dataset."""

    data = yaml.safe_load(path.read_text(encoding="utf-8"))
    if not isinstance(data, list):
        raise ValueError(f"{path} must be a YAML list of questions")

    questions = [Question.model_validate(q) for q in data]

    if len(questions) != _EXPECTED_COUNT:
        raise ValueError(
            f"Expected {_EXPECTED_COUNT} questions, got {len(questions)}"
        )

    counts: dict[str, int] = {c: 0 for c in _EXPECTED_CATEGORIES}
    ids: set[str] = set()
    for q in questions:
        if q.id in ids:
            raise ValueError(f"Duplicate question id: {q.id}")
        ids.add(q.id)
        counts[q.category] = counts.get(q.category, 0) + 1

    for cat, n in counts.items():
        if n != _EXPECTED_PER_CATEGORY:
            raise ValueError(
                f"Category {cat} has {n} questions, expected {_EXPECTED_PER_CATEGORY}"
            )

    return questions

```python
"""Claude-as-judge for RAG eval.

Uses the Anthropic API directly (NOT the Gemini key pool) so judge failures
can never cause generation failures and vice versa.
"""

from __future__ import annotations

import json
import os
import re
from typing import Any

from .types import AxisScore, JudgeResult, Question

_PROMPT = """You are scoring an answer produced by a retrieval-augmented chatbot.

# Question
{question}

# Retrieved chunks the chatbot saw
{chunks_block}

# Expected zettel ids (ground-truth citation targets)
{expected_ids}

# Expected behavior
{expected_behavior}

# Chatbot's answer
{answer}

# Rubric (score each axis 0-5 integer)
- faithfulness: are ALL claims in the answer grounded in the retrieved chunks? 5 = perfectly grounded, 0 = fabricated.
- relevance: does the answer directly address the question? 5 = laser-focused, 0 = off-topic.
- completeness: given what the chunks contain, does the answer cover the key facts? 5 = exhaustive, 0 = misses everything.
- citation_accuracy: does the answer cite the correct zettel(s) vs expected_zettel_ids? 5 = all expected ids cited, 0 = wrong or missing.

Special rules:
- If expected_behavior is "refuse" and the answer refuses (says it doesn't know / not in notes / no evidence), score ALL axes 5.
- If expected_behavior is "refuse" but the answer makes up content, faithfulness = 0 regardless of other axes.
- If retrieved_chunks is empty and the answer refuses, all axes = 5.

Return a strict JSON object, no preamble, no markdown fences, matching exactly:
{{
  "faithfulness": <0-5>,
  "relevance": <0-5>,
  "completeness": <0-5>,
  "citation_accuracy": <0-5>,
  "rationale": {{
    "faithfulness": "<one sentence>",
    "relevance": "<one sentence>",
    "completeness": "<one sentence>",
    "citation_accuracy": "<one sentence>"
  }}
}}
"""


def _format_chunks(chunks: list[dict[str, Any]]) -> str:
    if not chunks:
        return "(none)"
    lines: list[str] = []
    for i, c in enumerate(chunks[:20], start=1):
        node = c.get("node_id") or "?"
        text = str(c.get("text") or "")[:500].replace("\n", " ")
        lines.append(f"[{i}] node={node}: {text}")
    return "\n".join(lines)


def _extract_json(text: str) -> dict[str, Any]:
    text = text.strip()
    if text.startswith("```"):
        text = re.sub(r"^```(?:json)?\s*", "", text)
        text = re.sub(r"\s*```$", "", text)
    start = text.find("{")
    end = text.rfind("}")
    if start == -1 or end <= start:
        raise ValueError("no json object in text")
    return json.loads(text[start : end + 1])


class ClaudeJudge:
    """Anthropic Claude Sonnet judge. One retry on malformed output, then raise."""

    def __init__(self, *, client: Any, model: str = "claude-sonnet-4-5-20250929"):
        self._client = client
        self._model = model

    @classmethod
    def from_env(cls, *, model: str | None = None) -> "ClaudeJudge":
        from anthropic import AsyncAnthropic

        api_key = os.environ["ANTHROPIC_API_KEY"]
        client = AsyncAnthropic(api_key=api_key)
        return cls(client=client, model=model or "claude-sonnet-4-5-20250929")

    async def score(
        self,
        *,
        question: Question,
        answer: str,
        retrieved_chunks: list[dict[str, Any]],
    ) -> JudgeResult:
        prompt = _PROMPT.format(
            question=question.question,
            chunks_block=_format_chunks(retrieved_chunks),
            expected_ids=", ".join(question.expected_zettel_ids) or "(none)",
            expected_behavior=question.expected_behavior,
            answer=answer or "(empty)",
        )

        last_error: Exception | None = None
        for _ in range(2):
            try:
                msg = await self._client.messages.create(
                    model=self._model,
                    max_tokens=1024,
                    messages=[{"role": "user", "content": prompt}],
                )
                text = msg.content[0].text
                payload = _extract_json(text)
                scores = AxisScore(
                    faithfulness=int(payload["faithfulness"]),
                    relevance=int(payload["relevance"]),
                    completeness=int(payload["completeness"]),
                    citation_accuracy=int(payload["citation_accuracy"]),
                )
                rationale = {
                    k: str(v) for k, v in (payload.get("rationale") or {}).items()
                }
                return JudgeResult(scores=scores, rationale=rationale)
            except (ValueError, KeyError, json.JSONDecodeError) as exc:
                last_error = exc
                continue

        raise ValueError(f"could not parse judge response: {last_error}")
```

"""Rubric B — active 4-axis 0-5 scoring policy + loop advance decisions."""

from __future__ import annotations

from typing import Literal

from .types import AxisScore

_MEAN_GAIN_THRESHOLD = 0.5
_AXIS_REGRESSION_LIMIT = 1.0
_PLATEAU_THRESHOLD = 0.2
_HARD_ITER_CAP = 5


def _axis_means(scores: list[AxisScore]) -> dict[str, float]:
    if not scores:
        return {"faithfulness": 0.0, "relevance": 0.0, "completeness": 0.0, "citation_accuracy": 0.0}
    return {
        "faithfulness": sum(s.faithfulness for s in scores) / len(scores),
        "relevance": sum(s.relevance for s in scores) / len(scores),
        "completeness": sum(s.completeness for s in scores) / len(scores),
        "citation_accuracy": sum(s.citation_accuracy for s in scores) / len(scores),
    }


def _mean_total(scores: list[AxisScore]) -> float:
    if not scores:
        return 0.0
    return sum(s.total for s in scores) / len(scores)


def module_improved(previous: list[AxisScore], current: list[AxisScore]) -> bool:
    """Did the current iteration improve on the previous by rubric B rules?

    Rule: mean total rose ≥0.5 AND no axis regressed >1.0.
    """

    prev_mean = _mean_total(previous)
    curr_mean = _mean_total(current)
    if curr_mean - prev_mean < _MEAN_GAIN_THRESHOLD:
        return False

    prev_axes = _axis_means(previous)
    curr_axes = _axis_means(current)
    for axis, prev_val in prev_axes.items():
        if prev_val - curr_axes[axis] > _AXIS_REGRESSION_LIMIT:
            return False
    return True


def advance_decision(
    *,
    iter_in_module: int,
    recent_means: list[float],
) -> Literal["continue", "advance"]:
    """Should the loop advance to the next module?

    Advance iff:
    - iter_in_module >= HARD_ITER_CAP (5), OR
    - 2 consecutive iterations showed <0.2 mean gain (plateau).
    """

    if iter_in_module >= _HARD_ITER_CAP:
        return "advance"

    if len(recent_means) >= 3:
        d1 = recent_means[-1] - recent_means[-2]
        d2 = recent_means[-2] - recent_means[-3]
        if d1 < _PLATEAU_THRESHOLD and d2 < _PLATEAU_THRESHOLD:
            return "advance"

    return "continue"
```

"""Drive the eval run against /api/rag/adhoc."""

from __future__ import annotations

import asyncio
import os
import time
from datetime import datetime, timezone
from typing import Any, Protocol

import httpx

from .types import JudgeResult, Question, QuestionResult, RunReport

_NARUTO_EMAIL_DEFAULT = "naruto@zettelkasten.local"


class JudgeProtocol(Protocol):
    async def score(
        self,
        *,
        question: Question,
        answer: str,
        retrieved_chunks: list[dict[str, Any]],
    ) -> JudgeResult: ...


class SupabaseAuth:
    """Fetch and cache a Supabase JWT for Naruto (50 min TTL)."""

    def __init__(
        self,
        *,
        supabase_url: str,
        anon_key: str,
        email: str,
        password: str,
        http: httpx.AsyncClient,
    ):
        self._url = supabase_url.rstrip("/")
        self._anon = anon_key
        self._email = email
        self._password = password
        self._http = http
        self._cached_jwt: str | None = None
        self._expires_at: float = 0.0

    @classmethod
    def from_env(cls, *, http: httpx.AsyncClient) -> "SupabaseAuth":
        return cls(
            supabase_url=os.environ["SUPABASE_URL"],
            anon_key=os.environ["SUPABASE_ANON_KEY"],
            email=os.environ.get("NARUTO_EMAIL", _NARUTO_EMAIL_DEFAULT),
            password=os.environ["NARUTO_SUPABASE_PASSWORD"],
            http=http,
        )

    async def get_jwt(self) -> str:
        if self._cached_jwt and time.monotonic() < self._expires_at:
            return self._cached_jwt

        resp = await self._http.post(
            f"{self._url}/auth/v1/token?grant_type=password",
            headers={
                "apikey": self._anon,
                "Content-Type": "application/json",
            },
            json={"email": self._email, "password": self._password},
        )
        resp.raise_for_status()
        data = resp.json()
        self._cached_jwt = data["access_token"]
        self._expires_at = time.monotonic() + 50 * 60
        return self._cached_jwt


class EvalRunner:
    def __init__(
        self,
        *,
        http: httpx.AsyncClient,
        auth: SupabaseAuth,
        judge: JudgeProtocol,
        base_url: str,
    ):
        self._http = http
        self._auth = auth
        self._judge = judge
        self._base = base_url.rstrip("/")

    async def run(
        self,
        *,
        questions: list[Question],
        iteration: int,
        module_under_test: str,
        git_sha: str,
    ) -> RunReport:
        started = datetime.now(timezone.utc)
        results: list[QuestionResult] = []
        for q in questions:
            result = await self._run_one(q)
            results.append(result)
        finished = datetime.now(timezone.utc)
        return RunReport(
            iteration=iteration,
            module_under_test=module_under_test,
            started_at=started,
            finished_at=finished,
            git_sha=git_sha,
            rubric="b",
            results=results,
        )

    async def _run_one(self, question: Question) -> QuestionResult:
        try:
            jwt = await self._auth.get_jwt()
            resp = await self._ask_once(question, jwt)
            answer = str(resp.get("answer") or "")
            retrieved = list(resp.get("retrieved_chunks") or [])
            latency_ms = int(resp.get("latency_ms") or 0)
        except Exception as exc:
            return QuestionResult(
                question_id=question.id,
                category=question.category,
                kasten_id=question.kasten_id,
                question=question.question,
                answer="",
                retrieved_chunk_ids=[],
                latency_ms=0,
                error=f"ask: {exc}",
            )

        try:
            judge_result = await self._judge.score(
                question=question,
                answer=answer,
                retrieved_chunks=retrieved,
            )
        except Exception as exc:
            return QuestionResult(
                question_id=question.id,
                category=question.category,
                kasten_id=question.kasten_id,
                question=question.question,
                answer=answer,
                retrieved_chunk_ids=[str(c.get("id") or "") for c in retrieved],
                latency_ms=latency_ms,
                error=f"judge: {exc}",
            )

        return QuestionResult(
            question_id=question.id,
            category=question.category,
            kasten_id=question.kasten_id,
            question=question.question,
            answer=answer,
            retrieved_chunk_ids=[str(c.get("id") or "") for c in retrieved],
            latency_ms=latency_ms,
            judge=judge_result,
        )

    async def _ask_once(self, question: Question, jwt: str) -> dict[str, Any]:
        last_exc: Exception | None = None
        for attempt in range(2):
            try:
                resp = await self._http.post(
                    f"{self._base}/api/rag/adhoc",
                    headers={
                        "Authorization": f"Bearer {jwt}",
                        "Content-Type": "application/json",
                    },
                    json={
                        "message": question.question,
                        "stream": False,
                        "quality": "balanced",
                        "scope_filter": {},
                    },
                    timeout=60.0,
                )
                if resp.status_code >= 500:
                    raise RuntimeError(f"adhoc 5xx: {resp.status_code}")
                resp.raise_for_status()
                return resp.json()
            except Exception as exc:
                last_exc = exc
                if attempt == 0:
                    await asyncio.sleep(1.0 + attempt)
                    continue
                raise
        raise RuntimeError(f"unreachable: {last_exc}")
```

"""Render a RunReport into a human-readable markdown scorecard."""

from __future__ import annotations

import statistics
from collections import defaultdict

from .types import QuestionResult, RunReport


def _means_by_key(results: list[QuestionResult], key) -> dict[str, tuple[float, int]]:
    groups: dict[str, list[int]] = defaultdict(list)
    for r in results:
        if r.judge is None:
            continue
        groups[key(r)].append(r.judge.scores.total)
    return {k: (statistics.mean(v), len(v)) for k, v in groups.items()}


def render_scorecard(report: RunReport) -> str:
    scored = [r for r in report.results if r.judge is not None]
    errored = [r for r in report.results if r.error is not None]

    mean_total = report.mean_total()

    axis_means = {"faithfulness": 0.0, "relevance": 0.0, "completeness": 0.0, "citation_accuracy": 0.0}
    if scored:
        axis_means = {
            "faithfulness": statistics.mean(r.judge.scores.faithfulness for r in scored),
            "relevance": statistics.mean(r.judge.scores.relevance for r in scored),
            "completeness": statistics.mean(r.judge.scores.completeness for r in scored),
            "citation_accuracy": statistics.mean(r.judge.scores.citation_accuracy for r in scored),
        }

    by_category = _means_by_key(report.results, lambda r: r.category)
    by_kasten = _means_by_key(report.results, lambda r: r.kasten_id)

    latencies = [r.latency_ms for r in scored if r.latency_ms > 0]
    p50 = statistics.median(latencies) if latencies else 0
    p95 = statistics.quantiles(latencies, n=20)[18] if len(latencies) >= 20 else (max(latencies) if latencies else 0)

    worst = sorted(scored, key=lambda r: r.judge.scores.total)[:10]

    lines: list[str] = []
    lines.append(f"# RAG Eval Scorecard — iter {report.iteration:02d} — module `{report.module_under_test}`")
    lines.append("")
    lines.append(f"- **git sha:** `{report.git_sha}`")
    lines.append(f"- **rubric:** {report.rubric}")
    lines.append(f"- **window:** {report.started_at.isoformat()} → {report.finished_at.isoformat()}")
    lines.append(f"- **scored:** {len(scored)} / {len(report.results)} (errors: {len(errored)})")
    lines.append("")
    lines.append(f"## Mean total: **{mean_total:.2f} / 20**")
    lines.append("")
    lines.append("| axis | mean (0-5) |")
    lines.append("|---|---|")
    for axis, val in axis_means.items():
        lines.append(f"| {axis} | {val:.2f} |")
    lines.append("")
    lines.append("## Per-category")
    lines.append("")
    lines.append("| category | n | mean /20 |")
    lines.append("|---|---|---|")
    for cat, (mean, n) in sorted(by_category.items()):
        lines.append(f"| {cat} | {n} | {mean:.2f} |")
    lines.append("")
    lines.append("## Per-kasten")
    lines.append("")
    lines.append("| kasten | n | mean /20 |")
    lines.append("|---|---|---|")
    for kid, (mean, n) in sorted(by_kasten.items()):
        lines.append(f"| {kid} | {n} | {mean:.2f} |")
    lines.append("")
    lines.append("## Latency")
    lines.append("")
    lines.append(f"- p50: {p50:.0f} ms")
    lines.append(f"- p95: {p95:.0f} ms")
    lines.append("")
    lines.append("## Worst 10 answers")
    lines.append("")
    for r in worst:
        lines.append(f"### {r.question_id} ({r.category}) — {r.judge.scores.total}/20")

        lines.append("")

        lines.append("## Errors")
        lines.append("")
        for r in errored:
            lines.append(f"- `{r.question_id}`: {r.error}")
    return "\n".join(lines)

"""RAG module refactor loop controller.

One invocation = one iteration: run eval, score, write scorecard, update state,
decide advance. The /loop [5m] scheduler calls this repeatedly.

Reads:
- docs/superpowers/rag_eval/questions.yaml
- .rag_loop_state.json (in repo root)
Writes:
- docs/superpowers/rag_eval/reports/YYYY-MM-DD-iterNN-<module>.md
- .rag_loop_state.json (updated pointer)

Env required:
- SUPABASE_URL, SUPABASE_ANON_KEY
- NARUTO_SUPABASE_PASSWORD  (Naruto's password; never logged)
- ANTHROPIC_API_KEY
- RAG_EVAL_BASE_URL          (default: https://zettelkasten.in)
"""

from __future__ import annotations

import asyncio
import json
import os
import subprocess
import sys
from datetime import datetime, timezone
from pathlib import Path

import httpx

ROOT = Path(__file__).resolve().parents[2]
sys.path.insert(0, str(ROOT))

from website.features.rag_pipeline.evaluation.judge import ClaudeJudge  # noqa: E402
from website.features.rag_pipeline.evaluation.questions import load_questions  # noqa: E402
from website.features.rag_pipeline.evaluation.rubric_b import (  # noqa: E402

from website.features.rag_pipeline.evaluation.scorecard import render_scorecard  # noqa: E402

MODULE_ORDER = [
    "ingest",
    "retrieval+query",
    "rerank",
    "context",
    "generation+critic",
    "memory",
    "orchestrator",
]
GLOBAL_ITER_CAP = 20
SCORE_TARGET = 17.0
STATE_FILE = ROOT / ".rag_loop_state.json"
QUESTIONS_FILE = ROOT / "docs" / "superpowers" / "rag_eval" / "questions.yaml"
REPORTS_DIR = ROOT / "docs" / "superpowers" / "rag_eval" / "reports"
BASELINE_DIR = ROOT / "docs" / "superpowers" / "rag_eval"


def _default_state() -> dict:
    return {
        "current_module_idx": 0,
        "iter_in_module": 0,
        "total_iters": 0,
        "recent_means": [],
        "baseline_mean": None,
        "stopped": False,
        "stop_reason": None,
    }


def load_state() -> dict:
    if STATE_FILE.exists():
        return json.loads(STATE_FILE.read_text(encoding="utf-8"))
    return _default_state()


def save_state(state: dict) -> None:
    STATE_FILE.write_text(json.dumps(state, indent=2), encoding="utf-8")


def current_git_sha() -> str:
    return subprocess.check_output(
        ["git", "rev-parse", "--short", "HEAD"], cwd=ROOT, text=True
    ).strip()


async def wait_for_deploy(base_url: str, expected_sha: str, timeout_s: int = 300) -> bool:
    """Poll /api/health until git_sha matches or timeout."""
    deadline = asyncio.get_running_loop().time() + timeout_s
    async with httpx.AsyncClient(timeout=10.0) as http:
        while asyncio.get_running_loop().time() < deadline:
            try:
                resp = await http.get(f"{base_url.rstrip('/')}/api/health")
                data = resp.json()
                live = str(data.get("git_sha") or "")
                if live.startswith(expected_sha) or expected_sha.startswith(live):
                    return True
            except Exception:
                pass
            await asyncio.sleep(10.0)
    return False


async def run_iteration() -> int:
    state = load_state()
    if state["stopped"]:
        print(f"Loop already stopped: {state['stop_reason']}")
        return 0

    if state["total_iters"] >= GLOBAL_ITER_CAP:
        state["stopped"] = True
        state["stop_reason"] = f"hit global cap {GLOBAL_ITER_CAP}"
        save_state(state)
        print(state["stop_reason"])
        return 0

    module = MODULE_ORDER[state["current_module_idx"]]
    base_url = os.environ.get("RAG_EVAL_BASE_URL", "https://zettelkasten.in")
    sha = current_git_sha()

    print(f"[iter {state['total_iters']+1:02d}] module={module} sha={sha}")

    deployed = await wait_for_deploy(base_url, sha)
    if not deployed:
        print("Deploy did not match within timeout — aborting iteration")
        return 1

    questions = load_questions(QUESTIONS_FILE)

    async with httpx.AsyncClient(timeout=60.0) as http:
        auth = SupabaseAuth.from_env(http=http)
        judge = ClaudeJudge.from_env()
        runner = EvalRunner(http=http, auth=auth, judge=judge, base_url=base_url)
        report = await runner.run(
            questions=questions,
            iteration=state["total_iters"] + 1,
            module_under_test=module,
            git_sha=sha,
        )

    mean = report.mean_total()
    REPORTS_DIR.mkdir(parents=True, exist_ok=True)
    stamp = datetime.now(timezone.utc).strftime("%Y-%m-%d")
    report_path = REPORTS_DIR / f"{stamp}-iter{state['total_iters']+1:02d}-{module.replace('+', '_')}.md"
    report_path.write_text(render_scorecard(report), encoding="utf-8")
    (report_path.with_suffix(".json")).write_text(
        report.model_dump_json(indent=2), encoding="utf-8"
    )

    state["total_iters"] += 1
    state["iter_in_module"] += 1
    state["recent_means"].append(mean)
    state["recent_means"] = state["recent_means"][-5:]
    if state["baseline_mean"] is None:
        state["baseline_mean"] = mean
        (BASELINE_DIR / "baseline.json").write_text(
            report.model_dump_json(indent=2), encoding="utf-8"
        )

    print(f"[iter {state['total_iters']:02d}] mean_total={mean:.2f}/20 -> {report_path}")

    if mean >= SCORE_TARGET:
        state["stopped"] = True
        state["stop_reason"] = f"mean {mean:.2f} >= target {SCORE_TARGET}"
        save_state(state)
        print(state["stop_reason"])
        return 0

    decision = advance_decision(
        iter_in_module=state["iter_in_module"],
        recent_means=state["recent_means"][-3:],
    )
    if decision == "advance":
        state["current_module_idx"] += 1
        state["iter_in_module"] = 0
        if state["current_module_idx"] >= len(MODULE_ORDER):
            state["stopped"] = True
            state["stop_reason"] = "exhausted module order"

    save_state(state)
    return 0


def main() -> int:
    return asyncio.run(run_iteration())


if __name__ == "__main__":
    raise SystemExit(main())
```
